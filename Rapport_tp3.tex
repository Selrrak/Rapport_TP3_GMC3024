\subsubsection{Équations}
\begin{equation}
	AR\% = Stretch\% * \mu 
\end{equation}
\[ AR\% = 5\% * 0,5 = 2,5\% \]

\begin{equation}
	WR = W - (\frac{AR\%}{100})*W 
\end{equation}
\[WR = 2,62 - (\frac{2,5\%}{100})*2,62 = 2,56 mm\] 


\begin{equation}
	SQ\% = (\frac{WR - F}{W})*100\% 
\end{equation}

\[SQ\% = (\frac{2,56 - 2}{2,62})*100\% = 21,16\%  \]


\[SQ\% = (\frac{2,56 - 2}{2,62})*100\% = 21,16\%  \]

	F = (\frac{BORE-PG}{2})

\begin{equation}
	Stretch\% = (\frac{PG -ID}{ID})*100\%

\[ ID = \frac{PG}{(\frac{Stretch\%}{100})+1} \]
